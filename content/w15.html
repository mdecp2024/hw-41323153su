("從 1 累加到 100")

# 階段 1: 計算從 1 到 50 的總和
stage_1 = sum(range(1, 51))

# 階段 2: 計算從 51 到 100 的總和
stage_2 = sum(range(51, 101))

# 最終結果：將兩個階段的總和加起來
total = stage_1 + stage_2

# 輸出每個階段的結果和最終總和
print(f"階段 1 (1 到 50) 的總和是: {stage_1}")
print(f"階段 2 (51 到 100) 的總和是: {stage_2}")
print(f"階段 1 和階段 2 的總和: {stage_1} + {stage_2} = {total}")
print(f"從 1 累加到 100 的總和是: {total}")

def addto(start, end):
    total = 0
    for i in range(start, end + 1):  # 使用 range 設定從 start 到 end (包含 end)
        total += i  # 累加每個數字
    return total  # 返回總和

# 調用函式
result = addto(1, 100)
print("從 1 累加到 100 的總和是:", result)

def add_only_even(start, end):
    # 初始化總和為 0
    total = 0
    
    # 遍歷從 start 到 end 的數字
    for i in range(start, end + 1):
        if i % 2 == 0:  # 判斷是否為偶數
            total += i  # 如果是偶數，則累加到 total
    
    return total  # 返回總和

# 調用函式，計算從 1 到 100 的偶數總和
result = add_only_even(1, 100)
print(f"從 1 加到 100 之間的偶數總和是: {result}")

def add_avoid_8(start, end):
    total = 0  # 初始化總和
    for i in range(start, end + 1):  # 遍歷從 start 到 end 的每個數字
        if '8' not in str(i):  # 檢查數字是否包含 '8'
            total += i  # 如果不包含 '8'，則加到總和
    return total  # 返回總和

# 調用函式，計算從 1 到 100 的累加總和，避開數字中包含 '8' 的數字
result = add_avoid_8(1, 100)
print(f"從 1 到 100 累加，避開數字包含 '8' 的總和是: {result}")

def add_avoid_8(start, end):
    total = 0  # 初始化總和變數
    for i in range(start, end + 1):  # 從 start 到 end（包含 end）
        if '8' in str(i) or '3' in str(i):  # 檢查數字中是否包含 '8' 或 '3'
            continue  # 如果包含 '8' 或 '3'，跳過這個數字
        total += i  # 如果不包含，則將數字加到總和
    return total  # 返回總和

# 呼叫函式並輸出結果
result = add_avoid_8(1, 100)
print("從 1 累加到 100 的總和，避開含有 8 和 3 的數字，是:", result)
