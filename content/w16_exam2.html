w16_exam2

(迷宮)

from random import shuffle, randrange
from browser import document, window, html

# 生成迷宫的函数
def make_maze(w = 32, h = 16):  # 增大迷宫的宽度和高度
    # 1. 初始化迷宫的墙壁和访问状态
    vis = []  # 记录每个格子是否被访问过
    path = []  # 记录有效路径的格子

    # 使用双重循环来创建 h 行，每行有 w 个 0，0表示未访问
    for i in range(h):
        row = []  # 创建一个空的行列表
        for j in range(w):
            row.append(0)  # 将 0 添加到当前行
        vis.append(row)

    # 初始化 hor 和 ver 为空列表
    hor = []
    ver = []

    # 2. 使用双重循环创建水平方向墙壁 (hor)
    for i in range(h + 1):
        row = []  # 创建一行水平方向墙壁
        for j in range(w):
            row.append("+--")  # 每一列的墙壁是 "+--"
        hor.append(row)  # 将这一行添加到 hor 中

    # 使用双重循环创建垂直方向墙壁 (ver)
    for i in range(h):
        row = []  # 创建一行垂直方向墙壁
        for j in range(w):
            row.append("|  ")  # 每一列的墙壁是 "|  "
        ver.append(row)  # 将这一行添加到 ver 中

    # 3. 定义递归函数，生成迷宫
    def walk(x, y):
        # 标记当前格子为已访问
        vis[y][x] = 1
        path.append((x, y))  # 记录这个格子是路径的一部分

        # 定义四个可能的移动方向：左、下、右、上
        directions = [(x - 1, y), (x, y + 1), (x + 1, y), (x, y - 1)]
        shuffle(directions)  # 随机打乱方向顺序

        # 对每一个随机打乱的方向进行探索
        for xx, yy in directions:
            # 如果新的位置在迷宫范围内且未访问过
            if 0 <= xx < w and 0 <= yy < h and vis[yy][xx] == 0:
                # 如果是左右方向的移动，去除垂直墙壁
                if xx == x:
                    hor[max(y, yy)][x] = "+  "
                # 如果是上下方向的移动，去除水平墙壁
                if yy == y:
                    ver[y][max(x, xx)] = "   "

                # 递归调用，继续探索新位置
                walk(xx, yy)

    # 随机选择起点开始生成迷宫
    start_x = randrange(1, w - 1)  # 确保起点不在墙壁上
    start_y = randrange(1, h - 1)  # 确保起点不在墙壁上
    walk(start_x, start_y)

    # 返回生成的迷宫墙壁、起点、终点和路径
    return hor, ver, start_x, start_y, path

# 绘制迷宫
def draw_maze():
    w = 32  # 迷宫宽度，增大为 32
    h = 16  # 迷宫高度，增大为 16
    hor, ver, start_x, start_y, path = make_maze(w, h)  # 生成迷宫墙壁和路径

    # 动态设置画布大小
    canvas = html.CANVAS(1024, 1024)  # 增大画布
    div = document["brython_div1"]
    div <= canvas
    ctx = canvas.getContext("2d")

    # 每个格子的尺寸
    cell_width = canvas.width / w
    cell_height = canvas.height / h

    # 设置绘图颜色和线宽
    ctx.strokeStyle = "blue"  # 设置线条颜色为蓝色
    ctx.lineWidth = 2  # 设置线宽为 2 像素

    # 绘制水平方向的墙壁
    for i in range(h + 1):
        for j in range(w):
            if hor[i][j] == "+--":
                ctx.beginPath()
                ctx.moveTo(j * cell_width, i * cell_height)
                ctx.lineTo((j + 1) * cell_width, i * cell_height)
                ctx.stroke()

    # 绘制垂直方向的墙壁
    for i in range(h):
        for j in range(w):
            if ver[i][j] == "|  ":
                ctx.beginPath()
                ctx.moveTo(j * cell_width, i * cell_height)
                ctx.lineTo(j * cell_width, (i + 1) * cell_height)
                ctx.stroke()

    # 确保最右边的垂直墙壁也被绘制
    for i in range(h):
        ctx.beginPath()
        ctx.moveTo((w) * cell_width, i * cell_height)
        ctx.lineTo((w) * cell_width, (i + 1) * cell_height)
        ctx.stroke()

    # 标示入口与出口
    # 入口 (绿色) -> 位置会在边缘，避免压到线
    ctx.fillStyle = "#FF0000"
    ctx.beginPath()
    ctx.arc(start_x * cell_width + cell_width / 2, start_y * cell_height + cell_height / 2, 5, 0, 2 * window.Math.PI)
    ctx.fill()
    ctx.fillStyle = "#FF0000"
    ctx.fillText("(31,53)", start_x * cell_width + cell_width / 2 - 5, start_y * cell_height + cell_height / 2 - 10)

    # 出口 (蓝色) -> 位置会在边缘
    exit_x = w - 2  # 确保出口不压到墙壁
    exit_y = h - 2  # 确保出口不压到墙壁
    ctx.fillStyle = "#930000"
    ctx.beginPath()
    ctx.arc(exit_x * cell_width + cell_width / 2, exit_y * cell_height + cell_height / 2, 5, 0, 2 * window.Math.PI)
    ctx.fill()
    ctx.fillStyle = "#930000"
    ctx.fillText("出口", exit_x * cell_width + cell_width / 2 - 5, exit_y * cell_height + cell_height / 2 - 10)

# 在页面加载时，绘制迷宫
draw_maze()
